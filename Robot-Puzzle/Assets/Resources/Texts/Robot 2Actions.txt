function action()
	while true do
		tbl = sense()
		if tbl["type"] == "Ore" then
			moveOre()
		elseif tbl["type"] == "Rock" then
			moveRock()
		else
			release()
		end
	end
end

function moveOre()
	move("south")
	move("north")
end

function moveRock()
	grab()
	move("east")
	release()
	move("west")
end
