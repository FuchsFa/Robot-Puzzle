number = 0
red = {1,3,7,9}
blue = {2,4,6,8}

function action()
	while true do
		tbl = sense()
		if tbl["type"] == "Ore" then
			release()
			processOre()
		else
			release()
		end
	end
end

function processOre()
	number = number + 1
	if has_value(red, number) then
		paintRed()
	elseif has_value(blue, number) then
		paintBlue()
	else
		dontPaint()
	end
	returnHome()
	if number>= 9 then
		number = 0
	end
end

function paintRed()
	deliver()
	turnLeft()
	turnRight()
	release()
end

function paintBlue()
	deliver()
	turnRight()
	turnLeft()
	release()
end

function dontPaint()
	deliver()
	release()
end

function deliver()
	grab()
	turnRight()
	turnRight()
	walk()
	turnLeft()
	walk()
	walk()
	walk()
end

function returnHome()
	turnRight()
	turnRight()
	walk()
	walk()
	walk()
	turnRight()
	walk()
end

function has_value (tab, val)
    for index, value in ipairs(tab) do
        if value == val then
            return true
        end
    end

    return false
end
